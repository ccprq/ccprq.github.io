Diziler

  • Diziler, aynı türdeki verilerin bir koleksiyonunu saklamak için kullanılan veri yapılarıdır. C#, dizilerin boyutunu ve türünü tanımlamak için kullanılan bir "veri tipi[]" sözdizimine sahiptir.

  • Kullanım şekilleri aşağıda verilmiştir. 
    
    class Program
    {
        static void Main(string[] args)
        {
            int[] sayilar = new int[5]; // 5 elemanlı bir tam sayı dizisi oluştur
            sayilar[0] = 10; // Dizinin ilk elemanına 10 değerini atama
            sayilar[1] = 20; // Dizinin ikinci elemanına 20 değerini atama
            sayilar[2] = 30; // Dizinin üçüncü elemanına 30 değerini atama
            sayilar[3] = 40; // Dizinin dördüncü elemanına 40 değerini atama
            sayilar[4] = 50; // Dizinin beşinci elemanına 50 değerini atama
            
            int[] sayilar2 = new int[] { 10, 20, 30, 40, 50 }; // Elemanları direkt olarak tanımlayarak dizi oluşturma

            int uzunluk2 = sayilar.Length; // Dizinin uzunluğunu al

            int ikinciEleman = sayilar2[1]; // Dizinin ikinci elemanını al

            // İki boyutlu diziler
            // İki boyutlu diziler adındanda anlaşılacağı üzere 2 boyutludur.

            int[,] dizi = new int[3, 3];
            
            // Bu dizinin eleman alanı şöyledir ; 

            //       ***
            //       ***
            //       ***

            // Yani 3 sütün, 3 satır şeklinde

            dizi[0, 0] = 1; // Sol üst köşesine 1 değerini atama
            dizi[1, 2] = 5; // İkinci satırının üçüncü sütununa 5 değerini atama

            //       1**
            //       **5
            //       ***

            int satirSayisi = dizi.GetLength(0); // Satir sayisini alma
            int sutunSayisi = dizi.GetLength(1); // Sutun sayisini alma
            
            Console.ReadLine();
        }
    }
